access modes:
'B' = lrvalue
'L' = lvalue
'N' = nvalue
'R' = rvalue

typeinfo:
[0-9_] = invalid
globalname = class instance, functor
'c' .. globalname = class definition
'f' .. list(args) .. ti(rettype) = function (value-returning)
'f' .. list(args) .. 'N' = function (non-value-returning)
'n' = number
's' = string
'v' = nil

list encoding:
#items .. concat(items)

identifier encoding:
#str .. str, if str starts with [A-Za-z]
#str .. '_' .. str, if str starts with [0-9_]

globalname encoding:
ident(part), concatenated

cast ranks:
0 = identity
1 = RL -> R
2 = RL -> L
10 = upcast, functor cast, number->long
30 = user-defined cast
