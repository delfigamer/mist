access modes:
'B' = lrvalue
'L' = lvalue
'N' = nvalue
'R' = rvalue

typeinfo:
'_' = invalid
globalname = class instance
'c' .. globalname = class definition
'f' .. list(args) .. ti(rettype) = function (value-returning)
'f' .. list(args) .. 'N' = function (non-value-returning)
'n' = number
's' = string
'v' = nil

list encoding:
#items .. concat(items)

identifier encoding:
#str .. str, if str starts with [A-Za-z]
#str .. '_' .. str, if str starts with [0-9_]

globalname encoding:
ident(part), concatenated with '_'
