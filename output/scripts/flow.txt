// n is concentration
// T is temperature
// C is capacity
// S is conductance

D = 1 / C
p = n T D
i = - S grad p
dn/dt = - div i

S T D = A // diffusivity
i = - A {dn/dx, dn/dy}
div i = - A {ddn/dxx, ddn/dyy}
dn/dt = A (ddn/dxx + ddn/dyy) // a heat equation

Heat kernel:
PHI[d,t] = (4 pi a t)^-1.5 e^(- d^2 / 4 a t)

0 <= x <= 1
n[x, 0] = x
N = int[0,x] n dx
dn/dt = A ddn/dxx
dN/dt = A dn/dx
